💡 Code Examples

Basic API Usage

#!/usr/bin/env python3
"""
Solān AI Analysis Toolkit - Basic Usage Example
"""

import asyncio
import aiohttp
from datetime import datetime

class SolanClient:
    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url

    async def run_bias_scan(self, text_input):
        """Run bias detection analysis"""
        async with aiohttp.ClientSession() as session:
            async with session.post(f"{self.base_url}/api/analyzer/run-bias-scan", json={"text": text_input}) as response:
                return await response.json()

    async def test_ethics(self, ai_name, scenario="decision_making"):
        """Test AI ethics with specific scenario"""
        data = {
            "ai_name": ai_name,
            "scenario": scenario,
            "difficulty": "medium"
        }

        async with aiohttp.ClientSession() as session:
            async with session.post(f"{self.base_url}/api/analyzer/check-alignment", json=data) as response:
                return await response.json()

# Usage example
async def main():
    client = SolanClient()

    # Run bias detection scan
    bias_result = await client.run_bias_scan("This is a sample text for analysis")
    print(f"Bias Detection: {bias_result['biases_detected']}")

    # Test ethics
    ethics_result = await client.test_ethics("TestAI")
    print(f"Ethics Score: {ethics_result['ethics_result']['moral_score']}")

if __name__ == "__main__":
    asyncio.run(main())
